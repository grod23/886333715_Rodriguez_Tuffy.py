from contacts import *


def main_menu(contact_list):
    choice = 0
    while choice != 7:
        print('***Tuffy Titan Contact Main Menu***\n\n 1. Print List\n 2. Add Contact\n 3. Modify Contact\n '
              '4. Delete Contact\n 5. Sort Contacts by First Name\n 6. Sort Contacts by Last Name\n 7. Exit Main Menu\n')
        try:
            choice = int(input('Choose an option: '))
            if choice == 1:
                print_list(contact_list)
            elif choice == 2:
                print_list(add_contact(contact_list))
            elif choice == 3:
                print_list((modify_contact(contact_list)))
            elif choice == 4:
                print_list(delete_contact(contact_list))
            elif choice == 5:
                print_list(sort_first_name(contact_list))
            elif choice == 6:
                print_list(sort_last_name(contact_list))
        except ValueError:
            print('Invalid Choice!')


contacts = [['Gabriel', 'Rodriguez'], ['Steve', 'Gomez'], ['Alice', 'Tree']]
main_menu(contacts)

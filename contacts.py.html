def print_list(contact_list):
    print('***Tuffy Contact List***\n')
    for index, contact in enumerate(contact_list):
        print(f' {index} {contact[0]} {contact[1]}\n')


def add_contact(contact_list):
    new_contact = []
    first_name = input('What is contacts first name? ')
    last_name = input('What is contacts last name? ')
    new_contact.append(first_name)
    new_contact.append(last_name)
    contact_list.append(new_contact)
    return contact_list


def modify_contact(contact_list):
    try:
        index = int(input('What is the index of the contact you wish to change? '))
    except ValueError:
        print('Invalid Index!')
        return contact_list
    new_contact = []
    first_name = input('What is the contacts first name? ')
    last_name = input('What is the contacts last name? ')
    new_contact.append(first_name)
    new_contact.append(last_name)
    contact_list[index] = new_contact
    return contact_list


def delete_contact(contact_list):
    length = len(contact_list)
    index = int(input('What is the index of the contact you want to remove? '))
    if index < 0 or index >= length:
        print('Invalid index!')
        return contact_list
    contact_list.remove(contact_list[index])
    return contact_list


def sort_first_name(contact_list):
    contact_list.sort(key=lambda contact: contact[0])
    return contact_list


def sort_last_name(contact_list):
    contact_list.sort(key=lambda contact: contact[1])
    return contact_list

import unittest
from contacts import *


class test_01_add_contact(unittest.TestCase):
    def test_list_int(self):
        contacts = [['Richard', 'Stallman']]
        add_contact(contacts)
        result = [['Richard', 'Stallman'], ['Steve', 'Jobs']]
        self.assertEqual(contacts, result)


class test_02_modify_contact(unittest.TestCase):
    def test_list_int(self):
        contacts = [['Richard', 'Stallman'], ['Bill', 'Gates']]
        modify_contact(contacts)
        result = [['Richard', 'Stallman'], ['Steve', 'Jobs']]
        self.assertEqual(contacts, result)


class test_03_delete_contact(unittest.TestCase):
    def test_list_int(self):
        contacts = [['Richard', 'Stallman'], ['Bill', 'Gates'], ['Steve', 'Jobs']]
        delete_contact(contacts)
        result = [['Richard', 'Stallman'], ['Steve', 'Jobs']]
        self.assertEqual(contacts, result)


class test_04_sort_first_name(unittest.TestCase):
    def test_list_int(self):
        contacts = [['Richard', 'Stallman'], ['Bill', 'Gates'], ['Steve', 'Jobs']]
        sort_first_name(contacts)
        result = [['Bill', 'Gates'], ['Richard', 'Stallman'], ['Steve', 'Jobs']]
        self.assertEqual(contacts, result)


class test_05_sort_last_name(unittest.TestCase):
    def test_list_int(self):
        contacts = [['Richard', 'Stallman'], ['Bill', 'Gates'], ['Steve', 'Jobs']]
        sort_last_name(contacts)
        result = [['Bill', 'Gates'], ['Steve', 'Jobs'], ['Richard', 'Stallman']]
        self.assertEqual(contacts, result)
